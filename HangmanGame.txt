    private void wordGenActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_wordGenActionPerformed
        JOptionPane.showMessageDialog(null, "After 7 wrong guesses you're dead."  );
        wordGen.setEnabled(false);
        
        
        ArrayList<String> list= new ArrayList<String>();
        list.add("hello");
        list.add("java");
        list.add("python");
        list.add("fortran");
        list.add("javascript");
        list.add("matlab");
        Random rand = new Random();//to randomly generate the words above
        int i = rand.nextInt(list.size());
        word = list.get(i);
                
        int j = word.length();
        
        charArr = word.toCharArray();
        lengthOfArray = charArr.length;
        label[0] = jLabel1;
        label[1] = jLabel3;
        label[2] = jLabel4;
        label[3] = jLabel5;
        label[4] = jLabel6;
        label[5] = jLabel7;
        label[6] = jLabel8;
        label[7] = jLabel9;
        label[8] = jLabel10;
        label[9] = jLabel11;
        
        for (int k = 0; k < lengthOfArray; k++) {
            label[k].setText(String.valueOf("X"));
        }
        for(int l=0; l < list.size(); l++) {
            list.set(l, list.get(l).toUpperCase());//to ignore case sensitivity
            
            }
    }//GEN-LAST:event_wordGenActionPerformed

    private void guessButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_guessButtonActionPerformed
        String str = guessText.getText();
       
        int i = 0;
        char ch = str.charAt(0);
        for (int m=0 ; m < charArr.length ; m++){
           if (ch == charArr[m]){
           label[m].setText(String.valueOf(ch));
           i++;
           check++;
           }
           guessText.setText("");
        }
        if (check == lengthOfArray){
            messageLabel.setText("WON");
            JOptionPane.showMessageDialog(null, "Congratulations you have won.Game over. ");
            guessButton.setEnabled(false);
        }
        if (i == 0){
        
         jLabel2.setText("Result");
         String k = "/hangman/hm"+wrongGuess+".png";
        jLabel12.setIcon(new javax.swing.ImageIcon(getClass().getResource(k))); 
        wrongGuess++;
        }
        
        if (wrongGuess==8){
            messageLabel.setText("LOST");
            JOptionPane.showMessageDialog(null, "You did not win. Game over. The word is "+word);
            guessButton.setEnabled(false);
        }
        numberOfWrong.setText("Incorrect guesses: " +(int)(wrongGuess-1));
        
    }//GEN-LAST:event_guessButtonActionPerformed
